187  template<class Obsvbl, class... ArgN>
188  static auto blocking_subscribe(const Obsvbl& source, bool do_rethrow, ArgN&&... an)
189  -> void {
190  std::mutex lock;
191  std::condition_variable wake;
192  std::exception_ptr error;
193 
194  struct tracking
195  {
196  ~tracking()
197  {
198  if (!disposed || !wakened) abort();
199  }
200  tracking()
201  {
202  disposed = false;
203  wakened = false;
204  false_wakes = 0;
205  true_wakes = 0;
206  }
207  std::atomic_bool disposed;
208  std::atomic_bool wakened;
209  std::atomic_int false_wakes;
210  std::atomic_int true_wakes;
211  };
212  auto track = std::make_shared<tracking>();
213 
214  auto dest = make_subscriber<T>(std::forward<ArgN>(an)...);
215 
216  // keep any error to rethrow at the end.
217  auto scbr = make_subscriber<T>(
218  dest,
219  [&](T t){dest.on_next(t);},
220  [&](std::exception_ptr e){
221  if (do_rethrow) {
222  error = e;
223  } else {
224  dest.on_error(e);
225  }
226  },
227  [&](){dest.on_completed();}
228  );
229 
230  auto cs = scbr.get_subscription();
231  cs.add(
232  [&, track](){
233  // OSX geting invalid x86 op if notify_one is after the disposed = true
234  // presumably because the condition_variable may already have been awakened
235  // and is now sitting in a while loop on disposed
236  wake.notify_one();
237  track->disposed = true;
238  });
239 
240  std::unique_lock<std::mutex> guard(lock);
241  source.subscribe(std::move(scbr));
242 
243  wake.wait(guard,
244  [&, track](){
245  // this is really not good.
246  // false wakeups were never followed by true wakeups so..
247 
248  // anyways this gets triggered before disposed is set now so wait.
249  while (!track->disposed) {
250  ++track->false_wakes;
251  }
252  ++track->true_wakes;
253  return true;
254  });
255  track->wakened = true;
256  if (!track->disposed || !track->wakened) abort();
257 
258  if (error) {std::rethrow_exception(error);}
259  }
